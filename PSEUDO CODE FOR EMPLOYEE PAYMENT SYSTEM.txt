start
	payroll system management
  number of employees: 10
  for employees from 1 to 10. do
  input employee id
  input employee name
  input department
  input basic salary
  input allowance
  input deductions

	if (basic salary < 0 OR allowance < 0 OR deductions < 0)
   display (invalid payroll record)
	else,
   gross salary = basic salary + allowance
   net salary = gross salary - deductions
	if (gross salary > deductions)
   display (ready for payment)
	else
   display (payment review required)
endif
endfor
   display payroll report
   save file as payroll_report.txt
end
